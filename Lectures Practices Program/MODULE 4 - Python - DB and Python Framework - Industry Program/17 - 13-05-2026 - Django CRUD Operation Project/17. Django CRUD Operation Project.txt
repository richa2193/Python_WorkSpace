Ex:

from .forms import *

def insert(request):
    if request.method=='POST':
        form=Userform(request.POST)
        if form.is_valid():
            form.save()
            print("Record Inserted!")
            return redirect('/')
        else:
            print(form.errors)
    return render(request,'insert.html')
