Step 22. views.py 
now in views.py add this code for fetch data.

Ex:
from .forms import *

def signup(request):
    if request.method=='POST':
        form=Signupform(request.POST)
        if form.is_valid():
            form.save()
            print("Signup Successfully!")
            return redirect('/')
        else:
            print(form.errors)
    return render(request,'signup.html')
